# PostgreSQL Schema V1

## 1. Purpose

Defines the V1 database schema for the agent database.

This database is separate from the webapp database. It stores agent-local identity references,
event history, derived projections, runtime traceability, and compact conversation continuity
state. It does not own platform auth or webapp user records.

## 2. Required Extensions

```sql
CREATE EXTENSION IF NOT EXISTS pgcrypto;
```

## 3. Identity

```sql
CREATE TABLE agent_user_identities (
    agent_user_id TEXT PRIMARY KEY,
    external_system TEXT NOT NULL,
    external_subject TEXT NOT NULL,
    email TEXT NULL,
    status TEXT NOT NULL DEFAULT 'active'
        CHECK (status IN ('active', 'disabled', 'deleted')),
    created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
    updated_at TIMESTAMPTZ NOT NULL DEFAULT now(),
    CONSTRAINT ux_agent_user_identities_external_identity
        UNIQUE (external_system, external_subject)
);

CREATE INDEX ix_agent_user_identities_email
ON agent_user_identities (email);
```

`preferred_name` and display profile data must live in `user_profile_projection.profile`, not in
technical identity.

## 4. Runtime Traceability

```sql
CREATE TABLE agent_turns (
    turn_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    conversation_id TEXT NOT NULL,
    user_id TEXT NOT NULL,
    input_text TEXT NOT NULL,
    normalized_text TEXT NULL,
    selected_node TEXT NULL,
    final_status TEXT NOT NULL DEFAULT 'completed'
        CHECK (final_status IN ('completed', 'failed', 'blocked', 'needs_user_response')),
    created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
    metadata JSONB NOT NULL DEFAULT '{}'::jsonb
);

CREATE INDEX ix_agent_turns_user_created
ON agent_turns (user_id, created_at);

CREATE INDEX ix_agent_turns_conversation_created
ON agent_turns (conversation_id, created_at);

CREATE INDEX ix_agent_turns_selected_node
ON agent_turns (selected_node);

CREATE TABLE node_runs (
    node_run_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    turn_id UUID NOT NULL,
    user_id TEXT NOT NULL,
    node_name TEXT NOT NULL,
    input JSONB NOT NULL,
    output JSONB NOT NULL,
    status TEXT NOT NULL CHECK (
        status IN (
            'completed',
            'failed',
            'skipped',
            'rerouted',
            'needs_clarification',
            'needs_confirmation',
            'safety_blocked'
        )
    ),
    created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
    metadata JSONB NOT NULL DEFAULT '{}'::jsonb
);

CREATE INDEX ix_node_runs_turn ON node_runs (turn_id);
CREATE INDEX ix_node_runs_user_created ON node_runs (user_id, created_at);
CREATE INDEX ix_node_runs_node_name ON node_runs (node_name);
```

No hard foreign key is required from `node_runs.turn_id` to `agent_turns.turn_id` in V1.

## 5. Event Store

```sql
CREATE TABLE user_events (
    event_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    event_seq BIGINT GENERATED BY DEFAULT AS IDENTITY UNIQUE NOT NULL,
    user_id TEXT NOT NULL,
    turn_id UUID NULL,
    event_type TEXT NOT NULL,
    aggregate_type TEXT NOT NULL,
    aggregate_id TEXT NOT NULL,
    occurred_at TIMESTAMPTZ NOT NULL,
    recorded_at TIMESTAMPTZ NOT NULL DEFAULT now(),
    source TEXT NOT NULL CHECK (source IN ('user', 'system', 'import', 'migration', 'test')),
    actor JSONB NOT NULL DEFAULT '{}'::jsonb,
    correlation_id TEXT NULL,
    causation_id TEXT NULL,
    idempotency_key TEXT NULL,
    schema_version INTEGER NOT NULL DEFAULT 1,
    payload JSONB NOT NULL,
    metadata JSONB NOT NULL DEFAULT '{}'::jsonb
);

CREATE INDEX ix_user_events_user_seq
ON user_events (user_id, event_seq);

CREATE INDEX ix_user_events_user_occurred_at
ON user_events (user_id, occurred_at);

CREATE INDEX ix_user_events_user_recorded_at
ON user_events (user_id, recorded_at);

CREATE INDEX ix_user_events_type
ON user_events (event_type);

CREATE INDEX ix_user_events_aggregate
ON user_events (aggregate_type, aggregate_id);

CREATE INDEX ix_user_events_turn
ON user_events (turn_id);

CREATE INDEX ix_user_events_payload_gin
ON user_events USING gin (payload);

CREATE UNIQUE INDEX ux_user_events_idempotency
ON user_events (user_id, idempotency_key)
WHERE idempotency_key IS NOT NULL;
```

`event_type` remains `TEXT`; event validation belongs in `docs/contracts/events/event-registry.yml`
and code-level registries, not a PostgreSQL enum.

## 6. Projector Offsets

```sql
CREATE TABLE projector_offsets (
    projector_name TEXT PRIMARY KEY,
    last_event_seq BIGINT NOT NULL DEFAULT 0,
    updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);
```

Each projector must update its own offset after processing events.

## 7. Projections

```sql
CREATE TABLE user_profile_projection (
    user_id TEXT PRIMARY KEY,
    profile JSONB NOT NULL DEFAULT '{}'::jsonb,
    restrictions JSONB NOT NULL DEFAULT '[]'::jsonb,
    preferences JSONB NOT NULL DEFAULT '[]'::jsonb,
    active_goal_id TEXT NULL,
    last_event_seq BIGINT NOT NULL DEFAULT 0,
    updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE TABLE constraint_projection (
    user_id TEXT PRIMARY KEY,
    hard_constraints JSONB NOT NULL DEFAULT '[]'::jsonb,
    soft_constraints JSONB NOT NULL DEFAULT '[]'::jsonb,
    safety_flags JSONB NOT NULL DEFAULT '[]'::jsonb,
    derived_from_event_seq BIGINT NOT NULL DEFAULT 0,
    updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE TABLE nutrition_status_projection (
    user_id TEXT PRIMARY KEY,
    recent_meals JSONB NOT NULL DEFAULT '[]'::jsonb,
    biometrics JSONB NOT NULL DEFAULT '{}'::jsonb,
    symptoms JSONB NOT NULL DEFAULT '[]'::jsonb,
    computed_metrics JSONB NOT NULL DEFAULT '{}'::jsonb,
    last_event_seq BIGINT NOT NULL DEFAULT 0,
    updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE TABLE planning_history_projection (
    user_id TEXT PRIMARY KEY,
    active_session_id TEXT NULL,
    active_plan_artifact_id TEXT NULL,
    active_goal_id TEXT NULL,
    revision_summary JSONB NOT NULL DEFAULT '[]'::jsonb,
    feedback_summary JSONB NOT NULL DEFAULT '[]'::jsonb,
    last_event_seq BIGINT NOT NULL DEFAULT 0,
    updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);
```

Projections are derived state. They must not become the source of truth.

## 8. Conversation Continuity

```sql
CREATE TABLE conversation_state_summaries (
    conversation_id TEXT PRIMARY KEY,
    user_id TEXT NOT NULL,
    summary_version TEXT NOT NULL,
    summary JSONB NOT NULL DEFAULT '{}'::jsonb,
    should_inject_next_turn BOOLEAN NOT NULL DEFAULT true,
    updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE INDEX ix_conversation_state_summaries_user_updated
ON conversation_state_summaries (user_id, updated_at);

CREATE TABLE pending_interaction_state (
    conversation_id TEXT PRIMARY KEY,
    user_id TEXT NOT NULL,
    pending_kind TEXT NOT NULL CHECK (
        pending_kind IN ('question', 'confirmation', 'choice', 'proposal')
    ),
    assistant_prompt TEXT NOT NULL,
    expected_user_response TEXT NOT NULL,
    resume_graph TEXT NULL,
    resume_node TEXT NULL,
    created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
    updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE INDEX ix_pending_interaction_state_user_updated
ON pending_interaction_state (user_id, updated_at);
```

`conversation_state_summaries` is compact conversational memory only. It must not replace
`user_events` or projection truth.
